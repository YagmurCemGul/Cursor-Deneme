## Integration Example

### Before (Old Code):
```typescript
const handleOptimize = async () => {
  setIsOptimizing(true);
  const result = await aiService.optimizeCV(cvData, jobDescription);
  setIsOptimizing(false);
};
```

**Problems:**
- No loading indicator
- No error handling
- No input validation
- Uses mock data if AI fails

### After (New Code):
```typescript
const handleOptimize = async () => {
  // 1. Validate inputs
  const validation = validateOptimizationInputs(cvData, jobDescription);
  
  if (!validation.isValid) {
    setError(formatValidationMessage(validation));
    return;
  }
  
  // 2. Show loading
  setIsOptimizing(true);
  setError(null);
  
  try {
    // 3. Call AI (throws on error)
    const result = await aiService.optimizeCV(cvData, jobDescription);
    setOptimizations(result.optimizations);
  } catch (error) {
    // 4. Show error with retry
    setError(error);
  } finally {
    setIsOptimizing(false);
  }
};
```

**In JSX:**
```tsx
{/* Status Indicator */}
<AIProviderStatus 
  language={language}
  onConfigure={() => setActiveTab('settings')}
/>

{/* Loading State */}
{isOptimizing && (
  <LoadingState 
    language={language}
    message="Optimizing your CV with AI..."
    size="large"
  />
)}

{/* Error Display */}
{error && (
  <ErrorDisplay 
    error={error}
    language={language}
    onRetry={handleOptimize}
    onDismiss={() => setError(null)}
  />
)}

{/* First Launch Onboarding */}
{showOnboarding && (
  <AIOnboarding 
    language={language}
    onComplete={(config) => {
      aiService.updateConfig(config);
      setShowOnboarding(false);
    }}
  />
)}
```
